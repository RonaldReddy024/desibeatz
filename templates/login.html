@app.route('/login', methods=['GET', 'POST']) def login(): if request.method == 'POST': # Get credentials from the form. username = request.form.get('username') password = request.form.get('password') # Debug logs to check form input. app.logger.debug("Login attempt with username: %s", username) # Check for missing fields. if not username or not password: flash('All fields are required.', 'danger') app.logger.debug("Missing username or password") return redirect(url_for('login')) # Retrieve the user record. user = User.query.filter_by(username=username).first() if user: app.logger.debug("User found: %s", username) else: app.logger.debug("No user found with username: %s", username) # Verify the provided password. if user and user.verify_password(password): login_user(user) flash('Login successful!', 'success') app.logger.debug("Password verified. Logging in user: %s", username) return redirect(url_for('index1_html')) else: flash('Invalid username or password.', 'danger') app.logger.debug("Invalid credentials for username: %s", username) return redirect(url_for('login')) # For GET requests, render the login page. return render_template('login.html')
